# Belong backend dependencies — local / on-device only (no cloud AI SDKs).
# Only direct, actually-used deps are declared; transitive deps (tokenizers,
# ctranslate2, scipy, tqdm, …) are pulled in by these automatically.

# --- API & backend ---
fastapi
uvicorn
pydantic
python-multipart        # FastAPI multipart parsing (the /transcribe upload)
requests                # local vLLM (Nemotron) + Toronto Open Data calls

# --- Memory / data ---
chromadb                # vector store (semantic memory index)
numpy                   # cosine similarity (consolidation), vision arrays

# --- On-device vision (face enroll + "Who is this?") ---
insightface             # RetinaFace detector + ArcFace embeddings (buffalo_l)
onnxruntime             # runs the InsightFace + Chroma ONNX models locally
opencv-python-headless  # image decoding for the vision pipeline
pillow                  # image decode / thumbnail + EXIF rotation (photos.py, vision.py)

# --- On-device speech (the voice loop) ---
faster-whisper          # local STT (auto-detects the spoken language)
piper-tts               # local TTS (one Piper voice per language)
langdetect              # picks the TTS voice / detects the reply language (audio.py)

# --- Web Push reminders ---
pywebpush               # sends medication / appointment notifications (reminders.py)

# --- Data scripts (scripts/fetch_toronto_data.py) ---
pandas

# --- Tests ---
pytest                  # backend test runner
httpx                   # FastAPI TestClient HTTP backend
